Understanding User Growth Analysis

User growth analysis is the process of measuring, analyzing, and optimizing how your user base expands over time. It helps businesses understand acquisition channels, conversion rates, and growth patterns to make data-driven decisions about marketing strategies and product development.

Key Metrics

  • User Acquisition Rate - The number of new users gained in a specific time period
  • Conversion Rate - The percentage of visitors who complete a desired action
  • Customer Acquisition Cost (CAC) - The cost associated with acquiring a new user
  • Growth Rate - The percentage increase in users over a specific time period
  • K-Factor - A metric that measures the viral growth of a product

Linear charts help visualize user growth over time, showing trends, seasonality, and the impact of marketing campaigns.

Python Code Example


import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Create DataFrame directly

# Rotate x-axis labels for better readability
plt.xticks(rotation=45)

plt.show()

# Calculate month-over-month growth rate

Linear Regression for Growth Prediction

Linear regression helps predict future user growth based on historical data, allowing for more accurate resource planning and goal setting.

Example Dataset

Month Number Month New Users
1 Jan 2023 1200
2 Feb 2023 1350
3 Mar 2023 1500
4 Apr 2023 1650
5 May 2023 1800
6 Jun 2023 2100

This table shows monthly user acquisition data with month numbers for regression analysis.

Python Code Example


import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.metrics import r2_score

# Create DataFrame directly
df = pd.DataFrame({
    'Month_Num': [1, 2, 3, 4, 5, 6],
    'Month': ['Jan 2023', 'Feb 2023', 'Mar 2023', 'Apr 2023', 'May 2023', 'Jun 2023'],
    'New_Users': [1200, 1350, 1500, 1650, 1800, 2100]
})

# Prepare data for linear regression
X = df['Month_Num'].values.reshape(-1, 1)  # Independent variable
y = df['New_Users'].values  # Dependent variable

# Create and fit the model
model = LinearRegression()
model.fit(X, y)

# Make predictions for existing data points
df['Predicted_Users'] = model.predict(X)

# Predict future months (7-12)
future_months = np.array(range(7, 13)).reshape(-1, 1)
future_predictions = model.predict(future_months)

# Create future months DataFrame
future_df = pd.DataFrame({
    'Month_Num': range(7, 13),
    'Month': ['Jul 2023', 'Aug 2023', 'Sep 2023', 'Oct 2023', 'Nov 2023', 'Dec 2023'],
    'Predicted_Users': future_predictions
})

# Visualize actual vs predicted values
plt.figure(figsize=(12, 6))
plt.scatter(df['Month_Num'], df['New_Users'], color='blue', label='Actual Data')
plt.plot(df['Month_Num'], df['Predicted_Users'], color='red', label='Fitted Line')
plt.scatter(future_df['Month_Num'], future_df['Predicted_Users'], color='green', label='Future Predictions')
plt.plot(future_df['Month_Num'], future_df['Predicted_Users'], color='green', linestyle='--')

# Add labels and title
plt.title('User Growth Prediction with Linear Regression', fontsize=16)
plt.xlabel('Month Number')
plt.ylabel('New Users')
plt.xticks(range(1, 13), df['Month'].tolist() + future_df['Month'].tolist(), rotation=45)
plt.legend()

# Calculate and display model metrics
r2 = r2_score(df['New_Users'], df['Predicted_Users'])
plt.annotate(f'R² = {r2:.3f}', xy=(0.05, 0.95), xycoords='axes fraction')
plt.annotate(f'Slope = {model.coef_[0]:.2f}', xy=(0.05, 0.90), xycoords='axes fraction')

plt.tight_layout()
plt.show()

# Display combined actual and predicted data
print("Historical and Predicted User Growth:")
result_df = pd.concat([
    df[['Month', 'New_Users', 'Predicted_Users']],
    future_df[['Month', 'Predicted_Users']].rename(columns={'Predicted_Users': 'New_Users'})
])
print(result_df)

Funnel Chart for Conversion Rates

Funnel charts visualize the user journey through conversion stages, helping identify drop-off points and optimization opportunities.

Example Dataset

Funnel Stage Users Conversion Rate
Website Visits 10000 100%
Sign-up Page Views 5000 50%
Started Registration 3000 30%
Completed Registration 2000 20%
First Action 1500 15%
Became Active User 1000 10%

This table shows the number of users at each stage of the conversion funnel and the overall conversion rate from the top of the funnel.

Python Code Example


import pandas as pd
import plotly.graph_objects as go
import matplotlib.pyplot as plt
import numpy as np

# Create DataFrame directly
df = pd.DataFrame({
    'Stage': ['Website Visits', 'Sign-up Page Views', 'Started Registration', 
              'Completed Registration', 'First Action', 'Became Active User'],
    'Users': [10000, 5000, 3000, 2000, 1500, 1000],
    'Conversion_Rate': [100, 50, 30, 20, 15, 10]
})

# Calculate stage-to-stage conversion rates
df['Stage_Conversion'] = [100] + [round(df['Users'][i] / df['Users'][i-1] * 100, 1) for i in range(1, len(df))]

# Create funnel chart with Plotly
fig = go.Figure(go.Funnel(
    y = df['Stage'],
    x = df['Users'],
    textposition = "inside",
    textinfo = "value+percent initial",
    opacity = 0.8,
    marker = {
        "color": ["#003f5c", "#2f4b7c", "#665191", "#a05195", "#d45087", "#f95d6a"],
        "line": {"width": [0, 1, 1, 1, 1, 1], "color": ["white"]}
    },
    connector = {"line": {"color": "royalblue", "dash": "dot", "width": 3}}
))

fig.update_layout(
    title = "User Conversion Funnel",
    font_size = 14,
    width = 800,
    height = 500
)

# Add annotations for stage-to-stage conversion rates
for i in range(1, len(df)):
    fig.add_annotation(
        x = df['Users'][i] + 500,
        y = df['Stage'][i],
        text = f"{df['Stage_Conversion'][i]}% from previous",
        showarrow = False,
        font = dict(size = 12)
    )

fig.show()

# Alternative with Matplotlib
plt.figure(figsize=(10, 6))
plt.bar(df['Stage'], df['Users'], color=['#003f5c', '#2f4b7c', '#665191', '#a05195', '#d45087', '#f95d6a'])
plt.xticks(rotation=45, ha='right')
plt.title('User Conversion Funnel')
plt.ylabel('Number of Users')
plt.tight_layout()

# Add conversion rate labels
for i, v in enumerate(df['Users']):
    plt.text(i, v + 200, f"{df['Conversion_Rate'][i]}%", ha='center')
    if i > 0:
        plt.text(i, v/2, f"{df['Stage_Conversion'][i]}%\nfrom prev", ha='center', color='white')

plt.show()

# Display the data with stage-to-stage conversion
print(df[['Stage', 'Users', 'Conversion_Rate', 'Stage_Conversion']])

Distribution Heatmap for User Demographics

Distribution heatmaps help visualize user demographics across multiple dimensions, such as location, device type, and acquisition channel.

Example Dataset

Location Desktop Mobile Tablet
North America 3500 4200 800
Europe 2800 3100 600
Asia 1900 5600 300
South America 900 2100 200
Africa 400 1800 100
Oceania 700 900 150

This table shows the distribution of users by geographic location and device type.

Python Code Example


import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Create DataFrame directly
df = pd.DataFrame({
    'Location': ['North America', 'Europe', 'Asia', 'South America', 'Africa', 'Oceania'],
    'Desktop': [3500, 2800, 1900, 900, 400, 700],
    'Mobile': [4200, 3100, 5600, 2100, 1800, 900],
    'Tablet': [800, 600, 300, 200, 100, 150]
})

# Set index for better heatmap
df_heatmap = df.set_index('Location')

# Calculate percentages by region
df_pct = df_heatmap.div(df_heatmap.sum(axis=1), axis=0) * 100
df_pct = df_pct.round(1)

# Create heatmap
plt.figure(figsize=(12, 8))
sns.heatmap(df_heatmap, annot=True, cmap="YlGnBu", fmt="d", linewidths=.5)
plt.title('User Distribution by Location and Device Type', fontsize=16)
plt.tight_layout()
plt.show()

# Create percentage heatmap
plt.figure(figsize=(12, 8))
sns.heatmap(df_pct, annot=True, cmap="YlGnBu", fmt=".1f", linewidths=.5, cbar_kws={'label': 'Percentage %'})
plt.title('Device Usage Percentage by Location', fontsize=16)
plt.tight_layout()
plt.show()

# Add channel dimension (example with organic vs paid)
channels = ['Organic', 'Paid']
multi_dim_data = []

for location in df['Location']:
    for device in ['Desktop', 'Mobile', 'Tablet']:
        base_value = df.loc[df['Location'] == location, device].values[0]
        # Distribute between organic and paid (60/40 split as an example)
        organic = int(base_value * 0.6)
        paid = base_value - organic
        multi_dim_data.append([location, device, 'Organic', organic])
        multi_dim_data.append([location, device, 'Paid', paid])

# Create multi-dimensional DataFrame
df_multi = pd.DataFrame(multi_dim_data, columns=['Location', 'Device', 'Channel', 'Users'])

# Pivot for multi-dimensional analysis
df_pivot = df_multi.pivot_table(
    index=['Location', 'Device'],
    columns='Channel',
    values='Users',
    aggfunc='sum'
).reset_index()

print("Multi-dimensional user distribution:")
print(df_pivot)

# Calculate total users by location
total_by_location = df_heatmap.sum(axis=1).sort_values(ascending=False)
print("\nTotal users by location:")
print(total_by_location)

# Calculate device preference by location
device_pct_by_location = df_pct.idxmax(axis=1)
print("\nPreferred device by location:")
print(device_pct_by_location)

A/B Testing for Conversion Rates

A/B testing helps compare the effectiveness of different versions of a feature or page by measuring conversion rates between experimental groups.

Example Dataset

Variant Users Conversions Conversion Rate
Control (A) 5000 400 8.0%
Variant B 4950 495 10.0%

This table shows the number of users, conversions, and conversion rates for two variants in an A/B test.

Python Code Example


import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import scipy.stats as stats
from statsmodels.stats.proportion import proportions_ztest, proportion_confint

# Create DataFrame directly
df = pd.DataFrame({
    'Variant': ['Control (A)', 'Variant B'],
    'Users': [5000, 4950],
    'Conversions': [400, 495],
    'Conversion_Rate': [8.0, 10.0]
})

# Calculate conversion rates (if not already provided)
df['Calculated_Rate'] = df['Conversions'] / df['Users'] * 100

# Perform statistical test
count = np.array([df['Conversions'][0], df['Conversions'][1]])
nobs = np.array([df['Users'][0], df['Users'][1]])

# Z-test for proportions
z_stat, p_value = proportions_ztest(count, nobs)
print(f"Z-statistic: {z_stat:.4f}")
print(f"P-value: {p_value:.4f}")
print(f"Statistically significant difference: {p_value < 0.05}")

# Calculate confidence intervals
ci_control = proportion_confint(count[0], nobs[0], alpha=0.05, method='normal')
ci_variant = proportion_confint(count[1], nobs[1], alpha=0.05, method='normal')

# Calculate relative improvement
relative_improvement = ((df['Conversion_Rate'][1] - df['Conversion_Rate'][0]) / 
                        df['Conversion_Rate'][0] * 100)

# Visualize results
plt.figure(figsize=(10, 6))
bars = plt.bar(df['Variant'], df['Conversion_Rate'], 
        yerr=[(df['Conversion_Rate'][0]-ci_control[0]*100, df['Conversion_Rate'][1]-ci_variant[0]*100),
              (ci_control[1]*100-df['Conversion_Rate'][0], ci_variant[1]*100-df['Conversion_Rate'][1])],
        capsize=10, color=['#1f77b4', '#ff7f0e'])

# Add data labels
for bar in bars:
    height = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2., height + 0.3,
            f"{height:.1f}%", ha='center', va='bottom')

plt.title('A/B Test Results: Conversion Rate Comparison', fontsize=16)
plt.ylabel('Conversion Rate (%)')
plt.ylim(0, max(df['Conversion_Rate']) * 1.3)  # Add space for error bars and text

# Add statistical significance annotation
if p_value < 0.05:
    plt.annotate('Statistically Significant (p < 0.05)', 
                xy=(0.5, df['Conversion_Rate'].max() + 1.5),
                xytext=(0.5, df['Conversion_Rate'].max() + 3),
                ha='center', va='center',
                bbox=dict(boxstyle='round,pad=0.5', fc='yellow', alpha=0.3))
else:
    plt.annotate('Not Statistically Significant (p ≥ 0.05)', 
                xy=(0.5, df['Conversion_Rate'].max() + 1.5),
                xytext=(0.5, df['Conversion_Rate'].max() + 3),
                ha='center', va='center',
                bbox=dict(boxstyle='round,pad=0.5', fc='lightgray', alpha=0.3))

# Add relative improvement
plt.annotate(f"Relative Improvement: {relative_improvement:.1f}%", 
            xy=(0.5, df['Conversion_Rate'].max() + 0.5),
            ha='center', va='center')

plt.tight_layout()
plt.show()

# Calculate sample size for future tests
from statsmodels.stats.power import TTestIndPower

# Parameters
effect_size = (df['Conversion_Rate'][1] - df['Conversion_Rate'][0]) / 100  # Convert to proportion
alpha = 0.05  # Significance level
power = 0.8   # Power (1 - probability of Type II error)

# Calculate sample size
analysis = TTestIndPower()
sample_size = analysis.solve_power(effect_size=effect_size, power=power, alpha=alpha)

print(f"\nRecommended sample size for future tests: {int(np.ceil(sample_size))} per variant")

Case Studies

E-commerce Platform Growth Optimization

Problem: An e-commerce platform was experiencing stagnant growth despite increased marketing spend.

Analysis: User growth analysis revealed that while acquisition numbers were strong, conversion rates from first-time visitors to registered users were low, particularly on mobile devices.

Solution: The team simplified the mobile registration process and implemented a targeted remarketing campaign for visitors who abandoned the signup flow.

Result: 42% increase in mobile conversions and 27% improvement in overall user growth rate within three months.
